第一次ADD:
git config --global user.name "你的用户名"
git config --global user.email "你的邮箱地址"  
git init
git add .
git commit -m "第一次提交"
git remote add origin https://github.com/cnweich/yourname.git
git branch -M main
git push -u origin main

DELETE:
git clone https://github.com/你的用户名/仓库名.git
cd 仓库名
------win -------
del /F /Q *
rd /S /Q .
----------linux--------
rm -rf *
git add .
git commit -m "删除仓库全部内容"
git push origin main
//git push origin main --force（这会把远程的历史全部覆盖！提交、记录、文件都会消失。）

UPDATE:
git status
git add .
git commit -m "your_name"
git push

OTHERS:
#fatal: unable to access 'https://github.com/cnweich/gra.git/': Send failure: Connection was reset
# 如果你的代理端口是 7890（常见端口）
git config --global http.proxy http://127.0.0.1:7890
git config --global https.proxy http://127.0.0.1:7890

# 强制推送会覆盖远程仓库的内容
git push -u origin main --force

# 使用 rebase 方式，保留远程和本地的所有提交
git pull --rebase origin main
git push -u origin main

# 1. 先拉取远程的更改
git pull origin main --allow-unrelated-histories
# 2. 如果有冲突，解决冲突后提交
# 3. 再推送
git push -u origin main
